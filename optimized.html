class MySortedList extends Mn.CollectionView {
  constructor(opts) {
    this.tagName = 'ul';
    this.childView = CommitView;
    this.invert = false;
    super(opts);
  }
  invertSort(fast) {
    this.reorderOnSort = fast;
    this.invert = !this.invert;
    this.resortView();
  }
  viewComparator(model) {
    return (this.invert ? -1 : 1 ) * model.get('msg').length;
  }
};
